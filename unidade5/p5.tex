Dado:
\begin{equation}
    P_{\theta}(a < U < b) = P_{\theta}\left(a < \frac{X}{\theta} < b\right) = P_{\theta}\left(\frac{1}{b} < \theta < \frac{1}{a}\right) = P_{\theta}\left(\frac{X}{b} < \theta < \frac{X}{a}\right) = 1 - \alpha
\end{equation}

Isto é:
\begin{equation}
    ic_{1-\alpha}(\theta) = \left( \frac{X}{b}, \frac{X}{a} \right)
\end{equation}

IC é o intervalo bilateral para $\theta$ com confiança $1 - \alpha$.

\textbf{Exemplo 5:} Sejam $X_1, \ldots, X_n$ uma a.g. de $X \sim U(0, \theta)$ para $\theta$ desconhecido. Encontre o estimador intervalo bilateral para $\theta$ com confiança de $1 - \alpha$.

\textbf{Solução:} A estatística $T(X) = X_{\min}$ é suficiente mínima para $\theta$ com densidade
\begin{equation}
    f_T(t) = \frac{dF_T(t)}{dt} = \frac{d\left(F_X(t)\right)^n}{dt} = n \left( \frac{t}{\theta} \right)^{n-1} \frac{1}{\theta}
\end{equation}

Logo:
\begin{equation}
    f_T(t) = \frac{n}{\theta^n} t^{n-1} \quad , \quad t \in (0, \theta)
\end{equation}

Note que $U = \frac{T}{\theta}$ tem densidade
\begin{equation}
    f_U(u) = \frac{dF_U(u)}{du} = \frac{dP\left( \frac{T}{\theta} \leq u \right)}{du} = \theta \cdot f_T(u\theta) = n u^{n-1}
\end{equation}

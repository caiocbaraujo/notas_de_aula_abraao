\begin{equation}
\begin{aligned}
g'(u) &= \frac{n}{2} u^{\frac{n}{2} - 1} (n u b)^{-\frac{n m}{2}} - u^{\frac{n}{2}} \left( \frac{n m}{2} \right) (n u b)^{-\frac{n m}{2} - 1} \\
&= (n u b)^{-\frac{n m}{2}} \cdot u^{\frac{n}{2} - 1} \left[ (n b) \frac{n}{2} - u \left( \frac{n m}{2} \right) \right] \\
&= \frac{1}{a^2} (n u b)^{-\frac{n m}{2}} \cdot u^{\frac{n}{2} - 1} \left[ n b - m n u \right]
\end{aligned}
\end{equation}

Dai, $g(u) \geq 0$ e
\[
g'(u) = 0 \quad \Rightarrow \quad u = \frac{n b}{m}.
\]

(ii) $\lim_{u \to 0} g(u) = 0$ e $\lim_{u \to \infty} g(u) = 0$

Logo, $g(u)$ é muito pequena quando $u = \frac{\hat{\sigma}_x^2}{\hat{\sigma}_y^2}$ é muito pequena ou muito grande.

\textbf{Tomar:}
\begin{equation}
\begin{aligned}
F_n(\hat{\alpha}_1, \hat{\alpha}_2) &= \frac{\frac{\hat{\sigma}_x}{\sigma_x}}{\frac{\hat{\sigma}_y}{\sigma_y}} 
= \frac{\frac{n-1}{n} \frac{S_x^2}{\sigma_x^2}}{\frac{m-1}{m} \frac{S_y^2}{\sigma_y^2}} 
= \frac{\frac{n-1}{n} S_x^2 / \sigma_x^2}{\frac{m-1}{m} S_y^2 / \sigma_y^2} \\
&= \frac{S_x^2 / \sigma_x^2}{S_y^2 / \sigma_y^2} \cdot \frac{m(n-1)}{n(m-1)}
\end{aligned}
\end{equation}

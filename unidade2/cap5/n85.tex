
\begin{equation}
\begin{aligned}
P_{\mu} \{ \mu \in J_{x} \} &= P_{\mu} \left\{ \mu \in \left( \bar{X} - 1.96 \sqrt{2}, \; \bar{X} + 1.96 \sqrt{2} \right) \right\} \\
&= P_{\mu} \left\{ \bar{X} - \frac{1.96}{\sqrt{2}} < \mu < \bar{X} + \frac{1.96}{\sqrt{2}} \right\} \\
&= P_{\mu} \left\{ (\bar{X} - \mu) \sqrt{2} \leq 1.96 \; \land \; (\bar{X} - \mu) \sqrt{2} \geq -1.96 \right\} \\
&= P_{\mu} \left\{ \left| (\bar{X} - \mu) \sqrt{2} \right| \leq 1.96 \right\} \\
&\quad \quad \bar{X} \sim N(0,1) \\
&= P_{\mu} \left\{ |Z| \leq 1.96 \right\} = 0.95
\end{aligned}
\end{equation}

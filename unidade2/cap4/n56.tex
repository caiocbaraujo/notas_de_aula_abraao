\begin{equation}
R_c = \left\{ x \in \mathcal{X} : \sqrt{n} \frac{\bar{x} - \mu_0}{\sigma} > z_\alpha \right\}
\end{equation}
\begin{equation}
= \left\{ x \in \mathcal{X} : Z(X) > z_\alpha \right\}
\end{equation}
em que $z_\alpha$ é tal que
\begin{equation}
P(Z > z_\alpha) = \alpha
\end{equation}



\begin{equation}
-\frac{1}{\theta^n} \left[ \theta^n - \theta_0^n (1 - \alpha) \right] = 1 - (1 - \alpha) \left( \frac{\theta_0}{\theta} \right)^n \quad (4.5.2.1)
\end{equation}

Vamos a completar a função poder de $\psi^*$ e mostrar que coincide com (4.5.2.1).

Defina 
\[
g(t) = \mathbb{E} \left[ \psi^*(x) \mid T(x) = t \right]
\]
que não depende de $\theta$ pois $t(x)$ é suficiente para $\theta$. Assim,
\[
\alpha = \mathbb{E}_{\theta_0} \left\{ \psi^*(x) \right\} = \mathbb{E}_{\theta_0} \left\{ \mathbb{E} \left[ \psi^*(x) \mid T \right] \right\} = \mathbb{E}_{\theta_0} \left\{ g(T) \right\}.
\]

Note que $\psi^*(x) = 1$ para $t > \theta_0$ implica em $g(t) = 1$ se $t > \theta_0$.

Assim, para $\theta > \theta_0$,
\begin{equation}
\mathbb{E}_{\theta} \left\{ \psi^*(x) \right\} = \int_{0}^{\theta_0} g(t) \, n \, t^{n-1} \theta^{-n} \, dt + \int_{\theta_0}^{\theta} g(t) \, n \, t^{n-1} \theta^{-n} \, dt
\end{equation}

\begin{equation}
= \left( \frac{\theta_0}{\theta} \right)^n \int_{0}^{\theta_0} g(t) \, n \, t^{n-1} \theta_0^{-n} \, dt + \int_{\theta_0}^{\theta} n \, t^{n-1} \theta^{-n} \, dt
\end{equation}

Uma vez que $g(t) = 1$ para $t \geq \theta_0$:

\begin{equation}
\left( \frac{\theta_0}{\theta} \right)^n \alpha + \left[ \left( \frac{\theta}{\theta} \right)^n - \left( \frac{\theta_0}{\theta} \right)^n \right] = \left( \frac{\theta_0}{\theta} \right)^n \alpha + 1 - \left( \frac{\theta_0}{\theta} \right)^n = 1 - (1 - \alpha) \left( \frac{\theta_0}{\theta} \right)^n
\end{equation}

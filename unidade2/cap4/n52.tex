\begin{equation}
\begin{aligned}
&= \left[ E_{\theta_1} \, \psi_{\Upsilon}(x) - k \, E_{\theta_0} \, \psi_{\Upsilon}(x) \right] \\
&\quad - \left[ E_{\theta_1} \, \psi_{\Upsilon}^*(x) - k \, E_{\theta_0} \, \psi_{\Upsilon}^*(x) \right] \\
&0 \leq \left\{ Q_{\Upsilon}(\theta_1) - k \, Q_{\Upsilon}(\theta_0) \right\} \\
&\quad - \left\{ Q_{\Upsilon}^*(\theta_1) - k \, Q_{\Upsilon}^*(\theta_0) \right\} \\
&= Q_{\Upsilon}(\theta_1) - Q_{\Upsilon}^*(\theta_1) - k \left[ Q_{\Upsilon}(\theta_0) - Q_{\Upsilon}^*(\theta_0) \right]
\end{aligned}
\tag{4.3.4}
\end{equation}

Note que $Q_{\Upsilon}(\theta_0) = \alpha$ e $Q_{\Upsilon}^*(\theta_0) \leq \alpha$ e, portanto,
\begin{equation}
Q_{\Upsilon}(\theta_0) - Q_{\Upsilon}^*(\theta_0) \geq 0
\end{equation}

A (4.3.4) pode ser reescrita como
\begin{equation}
Q_{\Upsilon}(\theta_1) - Q_{\Upsilon}^*(\theta_1) \geq k \left[ Q_{\Upsilon}(\theta_0) - Q_{\Upsilon}^*(\theta_0) \right] \geq 0
\end{equation}
